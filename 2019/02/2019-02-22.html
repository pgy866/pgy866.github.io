arr = .Range("a1:z" & .Range("a100000").End(xlUp).Row)

End With

On Error Resume Next


For i = 3 To UBound(arr)

  If Left(arr(i, 2), 1) * 1 < 10 And arr(i, 2) <> "" Then

    For j = 0 To Len(arr(i, 2)) - Len(Replace(arr(i, 2), "/", ""))

      str = Split(arr(i, 2), "/")(j)

       

      arr(i, j * 2 + 3) = Left(str, Len(str) - (LenB(StrConv(str, vbFromUnicode)) - Len(str)))

      arr(i, j * 2 + 4) = Right(str, LenB(StrConv(str, vbFromUnicode)) - Len(str))

    Next j

  End If

Next i

Sheets("数据处理后").Range("a1:az50000").Clear

Sheets("数据处理后").Range("a1").Resize(UBound(arr), 26) = arr
